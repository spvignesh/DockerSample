Docker Version - 2.0.0.2 (30215)

docker ps 													(Lists all containers)

docker stop <<containerid>>											(Stops the running container)
docker start <<containerid>>											(Starts the container)


docker inspect -f '{{.Name}} - {{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' $(docker ps -aq)	(Gets all containers with Service IPAddress)
docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' <<container name/id>>		(Gets the container with Service IPAddress)
docker inspect <<container name/id>>										(Inspects the container)
